The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

def getFiles(zipFile):
    '''Takes a zip file and returns a dictionary of filenames with nested dictionaries for images
    param: a file of type .zip
    return: a dicitonary of type {filename : PIL.Image} }
    '''
    #define dictionary
    file_dict = {}
    #using the small_img file as the object myZip
    with zipfile.ZipFile(zipFile) as myZip:
        #get a list of file names in myZip and assign that to zip_files
        zip_files = myZip.infolist()
        #for each file in zip_files,
        for file in zip_files:
            #open file
            img_file = Image.open(myZip.open(file))
            file_name = file.filename
            #convert to greyscale
            image = img_file.convert('L')
            #add to dicitonary
            file_dict[file_name] = image
    return file_dict

def getFaces(image):
    '''Takes an image file and displays a contact sheet or prints string depending on results
    :param image: a zipfile image
    Output: displays contact sheet PIL.Image or prints string if no image available
    '''
    faces = []
    face_imgs = []
    thumb_size = 128,128
    
    #convert to numpy array
    image = image.convert("RGB")
    image = np.array(image)
    #get array info for faces
    faces.append(face_cascade.detectMultiScale(image, 1.3, 6))
    #cut out the faces from the image
    for face in faces:
        for x,y,w,h in face:
            #define face within image param
            rect = (x,y,x+w,y+h)
            #convert image to pil so we can use crop
            face_img = Image.fromarray(image)
            face_img = face_img.crop(rect)
            face_img.thumbnail(thumb_size)
            #append to our list of face images
            face_imgs.append(face_img)  
    
    if len(face_imgs) <1:
        print("But there were no faces in that file!")
    else:
        #define our contact sheet
        contact_sheet = PIL.Image.new(face_imgs[0].mode, (face_imgs[0].width*5, face_imgs[0].height*int(np.ceil(len(face_imgs)/5))))
        x=0
        y=0
    
        #add face images to contact sheet
        for img in face_imgs : 
            contact_sheet.paste(img, (x, y) )
            if x + face_imgs[0].width == contact_sheet.width:
                x= 0
                y= y + face_imgs[0].height
            else:
                x = x + face_imgs[0].width
        display(contact_sheet)

def getWords(word, image):
    '''takes a string 'word' and an image file and looks for the word in the image
    param image: zipfile image
    return: boolean
    '''
    #open file
    img = image.convert('1')
    #get words from image
    words = pytesseract.image_to_string(img)
    #remove duplicates
    #words = list(set(words))
    #for word in words:
    #words = word.lower()
    if word in words:
        return True
    else:
        return False
